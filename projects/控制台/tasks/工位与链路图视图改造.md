### 6.5 链路图渲染(替换原 `renderGraph`,删除旧 `linePoints`)
```js
const NODE_W=142;
function cubic(x1,y1,x2,y2){ const dx=Math.max(46,Math.abs(x2-x1)*0.46); return {d:`M ${x1} ${y1} C ${x1+dx} ${y1}, ${x2-dx} ${y2}, ${x2} ${y2}`,c1x:x1+dx,c2x:x2-dx}; }
function cubicMid(x1,y1,c1x,c2x,x2,y2){ const t=.5,mt=.5; return {x:mt*mt*mt*x1+3*mt*mt*t*c1x+3*mt*t*t*c2x+t*t*t*x2, y:(y1+y2)/2}; }
function renderGraph(){
  const wrap=document.querySelector('#flowmap'); if(!wrap) return;
  AGENT_META.forEach(m=>ensureAgent(m.id));
  const W=wrap.clientWidth||900, H=wrap.clientHeight||560;
  const edges={};
  BASE_EDGES.forEach(([from,to])=>{ edges[edgeKey(from,to)]={from,to,action:inferAction(from,to,{type:'edge.take'}),base:true}; });
  Object.entries(graphEdges).forEach(([k,v])=>{ edges[k]=v; });
  const ctr=id=>{ const n=GRAPH_POS[id]; return {x:n.x/100*W, y:n.y/100*H}; };
  const paths=Object.entries(edges).map(([key,e])=>{
    const A=GRAPH_POS[e.from], B=GRAPH_POS[e.to]; if(!A||!B) return '';
    const a=ctr(e.from), b=ctr(e.to);
    let x1,x2; if(b.x>=a.x){ x1=a.x+NODE_W/2; x2=b.x-NODE_W/2; } else { x1=a.x-NODE_W/2; x2=b.x+NODE_W/2; }
    const y1=a.y, y2=b.y, cu=cubic(x1,y1,x2,y2), m=cubicMid(x1,y1,cu.c1x,cu.c2x,x2,y2), active=key===latestEdgeKey;
    return `<path class="edge-line ${active?'active':''}" d="${cu.d}" marker-end="url(#arw)"></path>`+
           `<text class="edge-label ${active?'active':''}" x="${m.x}" y="${m.y-4}">${esc(e.action)}</text>`;
  }).join('');
  const nodes=GRAPH_NODES.map(n=>{
    const m=META[n.id], a=ensureAgent(n.id), state=a.state||'idle', task=displayTask(a).replace(/^当前任务: /,'');
    return `<div class="fnode ${state} ${a.active?'active':''}" style="left:${n.x}%;top:${n.y}%;--ac:${m.accent}">
      <span class="port pin"></span><span class="port pout"></span>
      <div class="fhead"><span class="fico">${AV[n.id]||'🤖'}</span><span class="fname">${esc(m.label)}</span><span class="fdot"></span></div>
      <div class="fbody"><div class="fstate">${statusLabel(state)}</div><div class="ftask">${esc(task||'待命')}</div></div>
    </div>`;
  }).join('');
  wrap.innerHTML=`<svg class="edge-layer" width="100%" height="100%" viewBox="0 0 ${W} ${H}" preserveAspectRatio="none">
    <defs><marker id="arw" viewBox="0 0 10 10" refX="8.5" refY="5" markerWidth="7" markerHeight="7" orient="auto-start-reverse"><path d="M0 0 L10 5 L0 10 z" fill="currentColor"></path></marker></defs>
    ${paths}</svg>${nodes}`;
}
```
> 不变形要点:SVG `viewBox` 用容器**像素**尺寸(W×H)、`preserveAspectRatio="none"`;节点用百分比定位;连线端点 = 节点中心 ± `NODE_W/2`。
